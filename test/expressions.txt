# Call with predicate expression

filter(0..9, {# % 2 == 0})

==>

Expr(
  CallExpr(
    VarName,
    Arguments(
      RangeExpr(Number, Number),
      Predicate(
        BinaryExpr(
          BinaryExpr(Pointer, ArithmeticOp, Number),
          CompareOp,
          Number
        )
      )
    )
  )
)

==>

      )
    )
  )
)
